00049       // Cycle through parameters
00050       for (std::size_t param_num = 0; param_num < paramset_helper.getParamsetSize(); param_num++, current_mask >>= 1) {
00051          // If not acceptable, leave zero
00052          if (!(current_mask & storage.getAcceptable()))

00054          // Otherwise, set number to the number of exist under this parametrization
00055          for (StateID ID = 0; ID < product.getStateCount(); ID++) {
00056             auto succs = storage.getNeighbours(ID, true, current_mask);
00057             StateID max_BA = 0;

00122       // Cycle through the levels of the DFS procedure

00124          // Update markings from the previous round

00133             // For the parametrization cycle through transitions
00134             forEach(transitions[param_num], [&](std::pair<StateID, StateID> trans) {
00135                std::size_t divisor = markings[trans.first].exits[param_num]; // Count succesor
00136                // Add probabilities
00137                if (divisor)
00138                   markings[trans.second].next_prob[param_num] += markings[trans.first].current_prob[param_num] / divisor ;
00139             });
