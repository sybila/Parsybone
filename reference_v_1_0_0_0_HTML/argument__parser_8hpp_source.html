00157                                 // Attach the stream to the file
00158                                 input_stream.open(*argument, std::ios::in);
00159                                 if (input_stream.fail())
00160                                         throw std::runtime_error(std::string("Program failed to open an intput stream file: ").append(*argument));
00161                                 file_parsed = true;
00162          } else {
00163             throw std::runtime_error(std::string("Wrong argument on the input stream: ").append(*argument));
00164          }
00165                 }       
00166
