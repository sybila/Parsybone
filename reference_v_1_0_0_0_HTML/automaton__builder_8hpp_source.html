00078    std::vector<std::string> getAtoms(const std::string & label) const {
00079       std::vector<std::string> atoms;
00080       std::string atom;
00081 
00082       // Go through the label
00083       for (std::size_t pos = 0; pos < label.length(); pos++) {
00084          char ch = label[pos];
00085          if (!(isspace(ch) || isalpha(ch) || isdigit(ch) || ch == '<' || ch == '>' || ch == '=' || ch == '!' || ch == '|' || ch == '&' || ch == '(' || ch == ')'))
00086             throw std::runtime_error(std::string("String: ").append(label).append(" contains invalid character: ").append(&ch).c_str());
00087 
00088          // If the atom is not a logic symbol, add it to current atom, otherwise store the atom, if any
00089          if (!(ch == '!' || ch == '|' || ch == '&' || ch == '(' || ch == ')'))
00090             atom.push_back(ch);
00091          else if (!atom.empty() && atom.compare("tt") && atom.compare("ff"))
00092             atoms.push_back(std::move(atom));
00093       }
00094       // Store the last atom, if any (only if the atom is the formula itself)
00095       if (!atom.empty() && atom.compare("tt") && atom.compare("ff"))
00096          atoms.push_back(std::move(atom));

00105    std::vector<std::pair<SpecieID, std::set<std::size_t> > > getValues(const std::vector<std::string> atoms) const {
00106       std::vector<std::pair<SpecieID, std::set<std::size_t> > > values;
00107 
00108       for (auto atom = atoms.begin(); atom != atoms.end(); atom++) {
00109          // Find comparison operator
00110          std::size_t symbol = atom->find_first_of("<>=");
00111          if (symbol == std::string::npos)
00112             throw std::invalid_argument(std::string("No comparison operator found in the atom ").append(*atom));
00113 
00114          // Find ID of the specie mentioned in the atom
00115          SpecieID ID = model.findID(atom->substr(0, symbol));
00116          if (ID >= model.getSpeciesCount())
00117             throw std::invalid_argument(std::string("Invalid specie name in the atom ").append(*atom));
00118 
00119          // Find a value the specie is compared to
00120          std::size_t value;
00121          try {
00122             value = boost::lexical_cast<size_t, std::string>(atom->substr(symbol+1));
00123          } catch (boost::bad_lexical_cast e) {
00124             output_streamer.output(error_str, std::string("Error while parsing a propery subformula ").append(*atom).append(": ").append(e.what()));
00125             throw std::runtime_error("boost::lexical_cast<size_t, std::string>(atom->substr(0, symbol-1) failed");
00126          }
00127          if (value < model.getMin(ID) || value > model.getMax(ID))
00128             throw std::invalid_argument(std::string("Value of the specie out of specie range in the atom ").append(*atom));
00129 
00130          // Fill in all the values satisfying the atom
00131          if (atom->find("<") != std::string::npos)
00132             values.push_back(std::make_pair(ID, fillInterval(value, model.getMax(ID))));
00133          else if (atom->find("=") != std::string::npos)
00134             values.push_back(std::make_pair(ID, fillInterval(value, value)));
00135          else if (atom->find(">") != std::string::npos)
00136             values.push_back(std::make_pair(ID, fillInterval(model.getMin(ID), value)));
00137       }
00138       return values;
00139    }

00144    AllowedValues getAllowed(const std::string & label) const {
00145       // Get atoms of the lable
00146       auto atoms = getAtoms(label);
00147       // Decide in which activation levels of species those atoms are true
00148       auto values = getValues(atoms);
