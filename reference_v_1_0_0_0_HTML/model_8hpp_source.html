00145    void sortParameters() {
00146       // Check all species
00147       forEach(species,[&](ModelSpecie & specie){
00148          // Sort parameters of the specie
00149          std::sort(specie.parameters.begin(), specie.parameters.end(), [](const Parameter & a, const Parameter & b) -> bool {
00150             // Control number of active regulators - specie with less regulators is considered bigger
00151             auto a_it = a.first.rbegin(); auto b_it = b.first.rbegin();
00152             std::size_t on_a = 0; std::size_t on_b = 0;
00153             while (a_it != a.first.rend()){
00154                on_a += *a_it++;
00155                on_b += *b_it++;
00156             }
00157             if (on_a != on_b)
00158                return on_a < on_b; // A goes first if it has less regulators
00159 
00160             // If numbers of activer regulators are equal, bigger is the one who has regulators sorted on the left
00161             auto a_it2 = a.first.begin(); auto b_it2 = b.first.begin();
00162             while (a_it2 != a.first.end()){
00163                if (*a_it2 > *b_it2)
00164                   return true; // A goes first if it has mostleft context
00165                else if(*a_it2 < *b_it2)
00166                   return false;
00167                a_it2++; b_it2++;
00168             }
00169             throw std::runtime_error("Multiple context definitions detected.");
00170          });
00171       });
00172    }
