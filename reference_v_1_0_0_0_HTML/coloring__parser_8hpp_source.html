00070         void parseMask() {
00071                 // Read bytemaks
00072                 char * byteblock = new char [static_cast<std::size_t>(file_size)];
00073                 input_file.read(byteblock, file_size);
00074 
00075                 // Cycle through bytemasks and reforge them into Parameters
00076                 for (std::size_t shade_num = 0; shade_num < file_size / sizeof(Paramset); shade_num++) {
00077                         Paramset temp = 0;
00078                         // Read bytes and move them to the front of the Parameters, when possible
00079                         for (std::size_t byte_num = 0; byte_num < sizeof(Paramset); byte_num++) {
00080                                 temp <<= 8;
00081                                 char val = byteblock[shade_num*sizeof(Paramset) + byte_num];
00082                                 temp += static_cast<unsigned char>(val);
00083                         }
00084                         colors_sets.push_back(temp);
00085                 }
