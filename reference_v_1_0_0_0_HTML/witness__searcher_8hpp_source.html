00054    void storeTransitions(const Paramset which, bool initial, std::size_t depth) {
00055       std::vector<std::pair<StateID, StateID> > trans;  // Temporary storage for the transitions
00056 
00057       // Go from the end till the lastly reached node
00058       for (std::size_t step = 0; step < depth; step++) {
00059          trans.push_back(std::make_pair(path[step+1], path[step]));
00060          markings[path[step]].succeeded |= which; // Mark found for given parametrizations
00061       }
00062       markings[path[depth]].succeeded |= which;
00063 
00064       // Add transitions to the parametrizations that allow them
00065       Paramset marker = paramset_helper.getLeftOne();
00066       for (std::size_t param = 0; param < paramset_helper.getParamsetSize(); param++) {
00067          if (which & marker) {
00068             transitions[param].insert(trans.begin(), trans.end());

00082    void DFS(const StateID ID, Paramset paramset) {
00083       // Add the state to the path
00084       path[depth] = ID;
00085       if (depth > max_depth)
00086          throw std::runtime_error("Depth boundary overcome during the DFS procedure.");
00087 
00088       // If a way to the source was found, apply it as well
00089       if (product.isInitial(ID))
00090          storeTransitions(paramset, true, depth);
00091 
00092       // Remove those with Cost lower than this level of the search (meaning that nothing more that cycles would be found)
00093       paramset &= ~depth_masks[depth];
00094 
00095       // Remove parametrizations that already have proven to be used/useless
00096       for (std::size_t level = 1; level < depth && paramset; level++)
00097          paramset &= ~markings[ID].busted[level];
00098 
00099       // If this state already has proven to lie on a path to the source, add this possible successors
00100       // Note that this works correctly due to the fact, that parametrizations are removed form the BFS during the coloring once they prove acceptable
00101       Paramset connected = markings[ID].succeeded & paramset;
00102       if (connected)
00103          storeTransitions(connected, false, depth);
00104 
00105       paramset &= ~markings[ID].busted[depth];
00106       markings[ID].busted[depth] |= paramset; // Forbid usage of these parametrizations for depth levels as high or higher than this one
00107 
00108       // If there is anything left, pass it further to the predecessors
00109       if (paramset) {
00110          depth++;
00111 
00112          auto predecessors = std::move(storage.getNeighbours(ID, false, paramset)); // Get predecessors
00113          auto pred_label = std::move(storage.getMarking(ID, false, paramset)); auto label_it = pred_label.begin(); // Get its values
00114 
00115          for (auto pred = predecessors.begin(); pred != predecessors.end(); pred++, label_it++) {
00116             DFS(*pred, paramset & *label_it); // Recursive descent with parametrizations passed from the predecessor.
00117          }
00118 
00119          depth--;
00120       }
00121    }
