50  void iterateColor(std::vector<std::size_t> & color) const {
51  // If there is a posibility to increaset the value (from left to right), increase it and end, otherwise null it and continue
52  for (std::size_t value_num = 0; value_num < color.size(); value_num++) {
53  // Increase and end
54  if (color[value_num] < max_colors[value_num]) {
55  color[value_num]++;
56  return;
57  }
58  // Null
59  else {
60  color[value_num] = 0;
61  }
62  }
63  }

183  std::vector<std::pair<std::size_t, std::string> > getColors(Parameters result_parameters) const {
184  // Vector to fill
185  std::vector<std::pair<std::size_t, std::string> > colors;
186  std::vector<std::size_t> work_color = subcolor_nums;
187  // Change the order of values to: from right to left
188  result_parameters = swap(result_parameters, getParamsetSize() - static_cast<std::size_t>(parameter_end - parameter_begin));
189  // Store a mask for each color with just its bit on, other off
190  Parameters color_mask = 1 << ((parameter_end - parameter_begin) - 1);
191 
192  // Cycle through all round colors
193  for (ColorNum col_num = parameter_begin; col_num < parameter_end; col_num++) {
194  // Output current values
195  if (result_parameters % 2)
196  colors.push_back(std::make_pair(color_mask, createColorString(work_color)));
197 
198  // Increase values
199  color_mask >>= 1;
200  result_parameters >>= 1;
201  iterateColor(work_color);
202  }
203  return colors;
204  }
