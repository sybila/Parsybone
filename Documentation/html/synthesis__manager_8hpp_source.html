76  void doComputation() {
77  // Basic (initial) coloring
78  colorProduct(user_options.witnesses());
79 
80  // Store colored final vertices
81  std::vector<Coloring> final_states = std::move(storage.getColor(product.getFinalStates()));
82  // Get the actuall results by cycle detection for each final vertex
83  for (auto final_it = final_states.begin(); final_it != final_states.end(); final_it++) {
84  // For general property, there must be new coloring for each final state!
85  if (!none(final_it->second) && !user_options.timeSerie())
86  detectCycle(*final_it);
87 
88  // Store results from this final state
89  analyzer->storeResults(Coloring(final_it->first, storage.getColor(final_it->first)));
90  }
91  }

98  void colorProduct(const WitnessUse wits_use) {
99  // Assure emptyness
100  storage.reset();
101 
102  // Get initial coloring
103  Parameters starting;
104  if(coloring_parser.input())
105  starting = coloring_parser.getColors()[split_manager->getRoundNum()];
106  else
107  starting = split_manager->createStartingParameters();
108 
109  if (none(starting))
110  return;
111 
112  // Set all the initial states to initial color
113  for (auto init_it = product.getInitialStates().begin(); init_it != product.getInitialStates().end(); init_it++)
114  storage.update(starting, *init_it);
115 
116  // Schedule all initial states for updates
117  std::set<StateID> updates(product.getInitialStates().begin(), product.getInitialStates().end());
118 
119  // Start coloring procedure
120  BFS_reach = std::move(model_checker->startColoring(starting, updates, split_manager->getRoundRange(), wits_use));
121  }

128  void detectCycle(const Coloring & init_coloring) {
129  // Assure emptyness
130  storage.reset();
131 
132  // Sechedule nothing for updates (will be done during transfer in the next step)
133  model_checker->startColoring(init_coloring.first, init_coloring.second, split_manager->getRoundRange());
134  }

142 public:
146  SynthesisManager(const ProductStructure & _product, ColorStorage & _storage)
147  : structure(_product.getKS()), storage(_storage), automaton(_product.getBA()), product(_product) {
148  // Create classes that help with the synthesis
149  split_manager.reset(new SplitManager(product.getFunc().getParametersCount()));
150  model_checker.reset(new ModelChecker(product, storage));
151  analyzer.reset(new ColoringAnalyzer(product));
152  searcher.reset(new WitnessSearcher(*analyzer, storage, product));
153  output.reset(new OutputManager(*analyzer, product, *split_manager, *searcher));
154 
155  total_colors = 0;
156  BFS_reach.resize(getParamsetSize());
157  }

165  void doSynthesis() {
166  time_manager.startClock("coloring");
167 
168  // Do the computation for all the rounds
169  for (;split_manager->valid(); split_manager->increaseRound()) {
170  doPreparation();
171  doComputation();
172  doConclusion();
173  }
174 
175  time_manager.ouputClock("coloring");
