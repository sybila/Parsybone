13 // ProductBuilder creates the Product for coloring, based on automaton and PKS
14 // States of product are indexed as (BA_state_count * KS_state_ID + BA_state_ID) - e.g. if 3-state BA state ((1,0)x(1)) would be at position 3*1 + 1 = 4

41  std::vector<StateID> getReachableBA(const StateID KS_ID, const StateID BA_ID) const {
42  // Vector to store them
43  std::vector<StateID> reachable;
44  // Cycle through all the transitions
45  for (std::size_t trans_num = 0; trans_num < automaton.getTransitionCount(BA_ID); trans_num++) {
46  // Check the transitibility
47  if (automaton.isTransitionFeasible(BA_ID, trans_num, structure.getStateLevels(KS_ID)))
48  reachable.push_back(automaton.getTargetID(BA_ID, trans_num));
49  }
50  return reachable;
51  }

71  void markStates() {
72  for (std::size_t ba_state_num = 0; ba_state_num < automaton.getStateCount(); ba_state_num++) {
73  // Insert the state if it is an initial state
74  if (ba_state_num == 0) {
75  for (std::size_t ks_state_num = 0; ks_state_num < structure.getStateCount(); ks_state_num++) {
76  product.initial_states.push_back(product.getProductID(ks_state_num, ba_state_num));
77  }
78  }
79  // Insert the state if it is a final state
80  if (automaton.isFinal(ba_state_num)) {
81  for (std::size_t ks_state_num = 0; ks_state_num < structure.getStateCount(); ks_state_num++) {
82  product.final_states.push_back(product.getProductID(ks_state_num, ba_state_num));
83  }
84  }
85  }

94  void createProductStates(const StateID KS_ID, const StateID BA_ID) {
95  // Add this state
96  product.addState(KS_ID, BA_ID, automaton.isInitial(BA_ID), automaton.isFinal(BA_ID), structure.getStateLevels(KS_ID));
97  const StateID ID = product.getProductID(KS_ID, BA_ID);
98 
99  // Get all possible BA targets
100  auto BA_targets = std::move(getReachableBA(KS_ID, BA_ID));
101 
102  // Add all the transitions possible
103  for (std::size_t KS_trans = 0; KS_trans < structure.getTransitionCount(KS_ID); KS_trans++) {
104  // Get transition data
105  const StateID KS_target_ID = structure.getTargetID(KS_ID, KS_trans);
106  const std::size_t step_size = structure.getStepSize(KS_ID, KS_trans);
107  const std::vector<bool> transitive_values = structure.getTransitive(KS_ID, KS_trans);
108 
109  // Add transition for all allowed targets
110  for (auto BA_traget_it = BA_targets.begin(); BA_traget_it != BA_targets.end(); BA_traget_it++) {
111  // Compute target position
112  const StateID target = product.getProductID(KS_target_ID, *BA_traget_it);
113  // Store the transition
114  product.addTransition(ID, target, step_size, transitive_values);
115  }
116  }
117  }

132  void buildProduct() {
133  // count product states
134  const std::size_t states_count = structure.getStateCount() * automaton.getStateCount();
135 
136  // Start storage
137  createEmptyStorage(states_count);
138 
139  // Add transitions
140  for (std::size_t KS_ID = 0; KS_ID < structure.getStateCount(); KS_ID++) {
141  for (std::size_t BA_ID = 0; BA_ID < automaton.getStateCount(); BA_ID++) {
142  createProductStates(KS_ID, BA_ID);
143  }
144  }
145 
146  // Create final and intial states vectors
147  markStates();
148  }
