50  std::vector<bool> getMask(std::string mask_string) const {
51  // Vector that will hold the mask instead of the string
52  std::vector<bool> mask;
53  // For all characters in the mask
54  std::for_each(mask_string.begin(), mask_string.end(),[&mask](char ch) {
55  // Check correctness of the symbols
56  if (ch != '0' && ch !='1')
57  throw std::runtime_error("Error occured while parsing a regulation. It seems that you have entered value other than 0 or 1 in the mask.");
58  // Push the value to the vector
59  try {
60  mask.push_back(boost::lexical_cast<bool, char>(ch));
61  } catch (boost::bad_lexical_cast e) {
62  output_streamer.output(error_str, std::string("Error occured while parsing a regulation: ").append(e.what()));
63  throw std::runtime_error("boost::lexical_cast<size_t, char>(temp_attr->value()) failed");
64  }
65  });
66  return mask;
67  }

148  template <class returnType>
149  bool getAttribute(returnType & requested_data, const rapidxml::xml_node<> * const current_node, const char* attribute_name, bool mandatory = true) const {
150  rapidxml::xml_attribute<> *temp_attr = 0;
151  // Try to get the attribute
152  temp_attr = current_node->first_attribute(attribute_name);
153  // Check if the attribute has been required
154  if (temp_attr == 0) {
155  if (mandatory)
156  throw std::runtime_error(std::string("Parser did not found the mandatory attribute ").append(attribute_name));
157  else
158  return false;
159  }
160  else {
161  // Try to convert attribute into requested data type
162  try {
163  requested_data = boost::lexical_cast<returnType, char*>(temp_attr->value());
164  } catch (boost::bad_lexical_cast e) {
165  output_streamer.output(error_str, std::string("Error while parsing an attribute ").append(attribute_name).append(": ").append(e.what()));
166  throw std::runtime_error("boost::lexical_cast<returnType, char*>(temp_attr->value()) failed");
167  }
168  }
169  return true;
170  }

178  void parseInteractions(const rapidxml::xml_node<> * const specie_node, size_t specie_ID) const {
179  rapidxml::xml_node<> *interaction;
180  // Interaction data
181  std::size_t source; std::size_t threshold; std::string label; EdgeConstrain constrain; bool observable;
182 
183  // Step into INTERACTIONS tag
184  interaction = getChildNode(specie_node, "INTERACTIONS");
185 
186  // Step into first INTER tag
187  interaction = getChildNode(interaction, "INTER");
188 
189  while (true) { // End when the current node does not have next sibling (all INTER tags were parsed)
190  // Get source ID and conver to integer.
191  getAttribute(source, interaction, "source");
192  // Get threshold and conver to integer.
193  getAttribute(threshold, interaction, "threshold");
194  // Get an edge label
195  if (!getAttribute(label, interaction, "label", false))
196  label = "";
197  // Convert label into an edge constrain
198  constrain = readConstrain(label);
199  // Get observable attribute
200  getAttribute(observable, interaction, "observ", false);
201 
202  // Add a new interaction to the specified target
203  model.addInteraction(source, specie_ID, threshold, constrain, observable);
204 
205  // Continue stepping into INTER tags while possible
206  if (interaction->next_sibling("INTER"))
207  interaction = interaction->next_sibling("INTER");
208  else break;
209  }
210  }

244  void parseSpecies(const rapidxml::xml_node<> * const structure_node) const {
245  rapidxml::xml_node<> *specie;
246  // Specie data
247  std::string name; size_t max; size_t basal;
248 
249  // Step into first SPECIE tag
250  specie = getChildNode(structure_node, "SPECIE");
251 
252  while (true) { // End when the current node does not have next sibling (all SPECIES tags were parsed)
253  // Get name of the specie.
254  getAttribute(name, specie, "name");
255  // Get max value and conver to integer.
256  getAttribute(max, specie, "max");
257  // Get basal value and conver to integer.
258  getAttribute(basal, specie, "basal");
259 
260  // Create a new specie
261  size_t specie_ID = model.addSpecie(name, max, basal);
262 
263  // Get all the interactions of the specie and store them to the model.
264  parseInteractions(specie, specie_ID);
265  // Get all the regulations of the specie and store them to the model.
266  parseRegulations(specie, specie_ID);
267 
268  // Continue stepping into SPECIE tags while possible
269  if (specie->next_sibling("SPECIE"))
270  specie = specie->next_sibling("SPECIE");
271  else break;
272  }

307  void parseStates(const rapidxml::xml_node<> * const automaton_node) const {
308  rapidxml::xml_node<> *state;
309  // State data
310  bool final;
311 
312  // Step into first SPECIE tag
313  state = getChildNode(automaton_node, "STATE");
314 
315  while (true) { // End when the current node does not have next sibling (all STATES tags were parsed)
316  // Find out whether the state is final
317  getAttribute(final, state, "final");
318 
319  // Create a new state
320  StateID ID = model.addState(final);
321 
322  // Get all the transitions of the state and store them to the model.
323  parseTransitions(state, ID);
324 
325  // Continue stepping into STATE tags while possible
326  if (state->next_sibling("STATE"))
327  state = state->next_sibling("STATE");
328  else break;
329  }

339  void createDocument() {
340  // Copy input data from stream into a string line by line
341  std::string input_line, input_data;
342  for (int lineno = 1; std::getline(*input_stream, input_line); ++lineno) {
343  input_data += input_line + "\n";
344  }
345 
346  // Copy data from String to newly created c-string
347  parsed_data.reset(new char[input_data.size() + 1]);
348  std::strcpy(parsed_data.get(), input_data.c_str());
349 
350  // Setup the parser with the c-string
351  try {
352  model_xml.parse<0>(parsed_data.get());
353  } catch (rapidxml::parse_error e) {
354  output_streamer.output(error_str, std::string("Error occured while trying to reconstruct xml document from the stream: ").append(e.what()));
355  throw std::runtime_error("rapidxml::xml_document<>.parse(char *) failed");
356  }
357  }

373  void parseInput() {
374  // Temporaries
375  rapidxml::xml_node<> *current_node;
376  float file_version;
377  std::string unspecified_regulations;
378 
379  // Create the parser
380  createDocument();
381 
382  // Step into first MODEL (main) tag
383  current_node = model_xml.first_node();
384  if (strcmp(current_node->name(), "MODEL") != 0)
385  throw std::runtime_error(std::string("Parsed found out that input does not start with the tag <MODEL> but with the <")
386  .append(current_node->name()).append("> instead").c_str());
387  // Find version number
388  getAttribute(file_version, current_node, "ver");
389 
390  // Parse Kripke Structure
391  output_streamer.output(verbose_str, "Started reading of the Kripke structure.");
392  current_node = getChildNode(current_node, "STRUCTURE");
393  getAttribute(unspecified_regulations, current_node, "unspec");
394  parseSpecies(current_node);
395 
396  // Parse Buchi Automaton
397  output_streamer.output(verbose_str, "Started reading of the Buchi automaton.");
398  current_node = getSiblingNode(current_node, "AUTOMATON");
399  parseStates(current_node);
400 
401  // Pass additional information
402  model.addAdditionalInformation(getUnspecType(unspecified_regulations), file_version);
403  };
