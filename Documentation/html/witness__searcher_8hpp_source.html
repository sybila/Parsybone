68  void countProb() {
69  // Probability of being in the first state
70  long double probability = 1.0;
71  // For each state divide probability by all its successors
72  for (std::size_t state_index = (lenght - 1); state_index > 1; state_index--) {
73  probability /= per_color_storage->getNeighbours(path[state_index], true, color_num).size();
74  }
75 
76  // Check if this initial state is already included, if so, add probability, otherwise create new state with probability of this path
77  auto state_prob = state_robustness.find(path[(lenght - 1)]);
78  if (state_prob == state_robustness.end())
79  state_robustness.insert(std::make_pair(path[(lenght - 1)], probability));
80  else
81  state_prob->second += probability;
82  }

99  void DFS(const StateID ID) {
100  // Add yourself to the path
101  path[lenght++] = ID;
102 
103  // If you are initial, print this path
104  if (product.isInitial(ID)) {
105  used_paths.insert(path[1]);
106  if (user_options.displayWintess())
107  displayWit();
108  if (user_options.robustness())
109  countProb();
110  }
111  else if (lenght < max_path_lenght) { // Continue DFS only if witness has still allowed lenght
112  // Get predecessors
113  const Neighbours & preds = per_color_storage->getNeighbours(ID, false, color_num);
114 
115  // List through predecessors
116  if (preds.size())
117  for (Neighbours::const_iterator pred_it = preds.begin(); pred_it != preds.end(); pred_it++) {
118  // Do not reuse paths
119  if (used_paths.find(*pred_it) != used_paths.end())
120  continue;
121  // If it was not used yet, continue in DFS
122  DFS(*pred_it);
123  }
124  }
125 
126  // Return
127  --lenght;
128  }

135  // Get synthetized colors
136  auto colors = analyzer.getColors();
137  per_color_storage.reset(new PerColorStorage(analyzer, storage, product));
138 
139  // Go through colors
140  for (color_num = 0; color_num < colors.size(); color_num++) {
141  // Display color if requested
142  if (user_options.coloring())
143  output_streamer.output(results_str, colors[color_num].second);
144 
145  // Get round values
146  std::size_t bit_num = getBitNum(colors[color_num].first);
147  max_path_lenght = (user_options.witnesses() == short_wit) ? BFS_reach[bit_num] : ~0;
148 
149  // Compute witnesses for given color from each final state, they are also displayed, if requested
150  for (auto final_it = product.getFinalStates().begin(); final_it != product.getFinalStates().end(); final_it++) {
151  // Restart values
152  lenght = 0;
153 
154  // Start search
155  DFS(*final_it);
156  }
157 
158  // Display robustness if requested
159  if (user_options.robustness()) {
160  double robustness = 0.0;
161  // For each possible initial state add robustness
162  forEach(state_robustness, [&robustness](const std::pair<StateID, double> & start) {
163  robustness += start.second;
164  });
165  // Divide by number of inital states and ouput
166  robustness /= state_robustness.size();
167  output_streamer.output(results_str, robustness);
168 
169  state_robustness.clear();
170  }
171 
172  used_paths.clear();
173  }
174  }
