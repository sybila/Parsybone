40  void computeSubspace() {
41  // Number of full rounds for all processes
42  rounds_count = all_colors_count / (user_options.procCount() * getParamsetSize());
43  ColorNum rest_bits = all_colors_count % (user_options.procCount() * getParamsetSize());
44  last_round_bits = getParamsetSize();
45 
46  // If there is some leftover, add a round
47  if (std::ceil(static_cast<double>(rest_bits) / static_cast<double>(getParamsetSize())) >= user_options.procNum()) {
48  rounds_count++;
49  // Pad last round
50  if ((rest_bits / getParamsetSize()) == (user_options.procNum() - 1))
51  last_round_bits = rest_bits % getParamsetSize();
52  }
53 
54  // Get colors num for this process
55  process_color_count = (rounds_count - 1) * getParamsetSize() + last_round_bits;
56 
57  // Set positions for the round
58  setStartPositions();
59  // Compute number of full rounds
60 
61  // Check if it fits together with number from mask
62  if (coloring_parser.input())
63  if (coloring_parser.getColors().size() != rounds_count)
64  throw std::runtime_error(std::string("Rounds computed from bitmask: ").append(boost::lexical_cast<std::string>(coloring_parser.getColors().size()))
65  .append(" does not equal round count computed from model: ")
66  .append(boost::lexical_cast<std::string>(rounds_count).c_str()));
67  }

89  void setStartPositions() {
90  round_begin = (user_options.procNum() - 1) * getParamsetSize();
91  if (rounds_count > 0) {
92  if (rounds_count > 1)

103  void increaseRound() {
104  round_number++;
105  round_begin += (getParamsetSize() * user_options.procCount());
106  // For the last round we have to use a shorter range, if necessary, otherwise we use whole
107  if (lastRound())

110  round_end = round_begin + getParamsetSize();
