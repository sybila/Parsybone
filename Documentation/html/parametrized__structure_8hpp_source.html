13 // ParametrizedStructure stores states of the Kripke structure created from the model together with labelled transitions.
14 // Each transition contains a function that causes it with explicit enumeration of values from the function that are transitive.
15 // To easily search for the values in the parameter bitmask, step_size of the function is added
16 // - that is the value saying how many bits of mask share the the same value for the function.
17 // ParametrizedStructure data can be set only form the ParametrizedStructureBuilder object.
18 // Rest of the code can access the data only via constant getters - once the data are parse, model remains constant.

29 // NEW TYPES AND DATA:
31  // Storing a single transition to neighbour state together with its transition function
32  struct Transition {
33  StateID target_ID; // ID of the state the transition leads to
34  std::size_t step_size; // How many bits of a parameter space bitset is needed to get from one targe value to another
35  std::vector<bool> transitive_values; // Which values from the original set does not allow a trasition and therefore removes bits from the mask.
36 
37  Transition(const std::size_t _target_ID, const std::size_t _step_size, std::vector<bool>&& _transitive_values)
38  : target_ID(_target_ID), step_size(_step_size), transitive_values(std::move(_transitive_values)) {}
39  };

41  // Simple state enriched with transition functions
42  struct State {

111  const std::string getString(const StateID ID) const {
112  std::string state_string = "(";
