156  template <class outputType>
157  const OutputStreamer & output(StreamType stream_type, const outputType & stream_data, const unsigned int trait_mask = 0) {
158  // Update stream
159  last_stream_type = stream_type;
160  switch (stream_type) {
161  case error_str:
162  // Start with !
163  *error_stream << "! ";
164  break;
165  case stats_str:
166  // Start with #
167  if (user_options.stats())
168  *verbose_stream << "# ";
169  break;
170  case verbose_str:
171  // Start with * - if requseted, remove previous line
172  if (user_options.verbose()) {
173  if (testTrait(rewrite_ln, trait_mask))
174  *verbose_stream << '\r';
175  *verbose_stream << "* ";
176  }
177  break;
178  }
179  // Continue with output, then return this object
180  return output(stream_data, trait_mask);
181  }

189  template <class outputType>
190  const OutputStreamer & output(const outputType & stream_data, const unsigned int trait_mask = 0) const {
191  // Pick the correct stream and pass the data - output only if requested
192  switch (last_stream_type) {
193  case error_str:
194  actualOutput(*error_stream, stream_data, trait_mask);
195  break;
196  case results_str:
197  actualOutput(*result_stream, stream_data, trait_mask);
198  break;
199  case stats_str:
200  if (user_options.stats())
201  actualOutput(*stats_stream, stream_data, trait_mask);
202  break;
203  case verbose_str:
204  if (user_options.verbose())
205  actualOutput(*verbose_stream, stream_data, trait_mask);
206  break;
207  }
208  return *this;
209  }

219  template <class outputType>
220  void actualOutput(std::ostream & stream, const outputType & stream_data, const unsigned int trait_mask) const {
221  // Add prefix stars
222  if (testTrait(important, trait_mask))
223  stream << "-- ";
224  // Actuall data
225  stream << stream_data;
226  // Add postfix stars
227  if (testTrait(important, trait_mask))
228  stream << " --";
229  // End of the line if not requested otherwise
230  if (!testTrait(no_newl, trait_mask))
231  stream << std::endl;
232  }
