44  const bool testRegulators(const std::vector<StateID> & source_species, const std::vector<std::vector<std::size_t> > & source_values, const Levels & state_levels) {
45  // List throught regulating species of the function
46  for (std::size_t regulator_num = 0; regulator_num < source_species.size(); regulator_num++) {
47  bool is_included = false; // Remains false if the specie level is not in allowed range
48  const StateID ID = source_species[regulator_num]; // real ID of the regulator
49 
50  // Does current level of the specie belongs to the levels that are required?
51  std::for_each(source_values[regulator_num].begin(), source_values[regulator_num].end(), [&]
52  (std::size_t feasible_level) {
53  if (feasible_level == state_levels[ID])
54  is_included = true;
55  });
56 
57  // If specie does not meet requirments, do not even continue.
58  if (!is_included) {
59  return false;
60  }
61  }
62  // Return true if all species passed.
63  return true;
64  }

74  const std::size_t getActiveFunction(const std::size_t specie_ID, const Levels & state_levels) {
75  // Source species that will be tested
76  const std::vector<std::size_t> & source_species = regulatory_functions.getSourceSpecies(specie_ID);
77 
78  // Cycle until the function is found
79  bool found = false;
80  for (std::size_t regul_num = 0; regul_num < regulatory_functions.getRegulationsCount(specie_ID); regul_num++) {
81  const auto source_vals = regulatory_functions.getSourceValues(specie_ID, regul_num);
82 
83  found = testRegulators(source_species, source_vals, state_levels);
84 
85  if (found)
86  return regul_num;
87  }
88  throw std::runtime_error("Active function in same state not found.");
89  }

107  std::vector<bool> fillTransitivityData(const Direction direction, const std::size_t current_specie_level, const std::vector<std::size_t> & possible_values){
108  // Vector to fill
109  std::vector<bool> transitive_values;
110  // Based on direction of the change create a mask of transitivity for all parameter values
111  if (direction == up_dir) {
112  std::for_each(possible_values.begin(), possible_values.end(),[current_specie_level, &transitive_values](const std::size_t value){
113  if (value < current_specie_level + 1) // If the value is not at least as high as the target
114  transitive_values.push_back(false);
115  else transitive_values.push_back(true);
116  });
117  }
118  else if (direction == stay_dir){
119  std::for_each(possible_values.begin(), possible_values.end(),[current_specie_level, &transitive_values](const std::size_t value){
120  if (value != current_specie_level) // If the value is not same as the target
121  transitive_values.push_back(false);
122  else transitive_values.push_back(true);
123  });
124  }
125  else if (direction == down_dir) {
126  std::for_each(possible_values.begin(), possible_values.end(),[current_specie_level, &transitive_values](const std::size_t value){
127  if (value > current_specie_level - 1) // If the value is not at least as low as the target
128  transitive_values.push_back(false);
129  else transitive_values.push_back(true);
130  });
131  }
132  return transitive_values;
133  }

148  const bool fillFunctions(const StateID ID, const std::size_t neighbour_index, const Levels & state_levels,
149  std::size_t & step_size, std::vector<bool> & transitive_values) {
150  // Get ID of the regulated specie
151  const std::size_t specie_ID = basic_structure.getSpecieID(ID, neighbour_index);
152 
153  // Find out which function is currently active
154  std::size_t function_num = getActiveFunction(specie_ID, state_levels);
155 
156  // Fill step size
157  step_size = regulatory_functions.getStepSize(specie_ID, function_num);
158 
159  // Fill data about transitivity using provided values
160  transitive_values = std::move(fillTransitivityData(basic_structure.getDirection(ID, neighbour_index), state_levels[specie_ID],
161  regulatory_functions.getPossibleValues(specie_ID, function_num)));
162 
163  // Check if there even is a transition
164  for (auto it = transitive_values.begin(); it != transitive_values.end(); it++) {
165  if (*it == true)
166  return true;
167  }
168  return false;
169  }

177  void addTransitions(const StateID ID, const Levels & state_levels) {
178  // Go through all the original transitions
179  for (std::size_t trans_num = 0; trans_num < basic_structure.getTransitionCount(ID); trans_num++) {
180  // Data to fill
181  StateID target_ID = basic_structure.getTargetID(ID, trans_num); // ID of the state the transition leads to
182  std::size_t step_size = 1; // How many bits of a parameter space bitset is needed to get from one targe value to another
183  std::vector<bool> transitive_values; // Which of possible are used (in this case)
184 
185  // Fill data about the transition and check if it is even feasible
186  if (fillFunctions(ID, trans_num, state_levels, step_size, transitive_values)) {
187  // Add the transition
188  structure.addTransition(ID, target_ID, step_size, std::move(transitive_values));
189  }
190  }
191  }
