55  bool testConstrains(bool & is_observable, const SpecieID ID, const std::size_t regul_num, const std::size_t inter_num, const std::vector<std::size_t> & subcolor) const {
56  // Get reference data
57  const std::vector<Model::Interaction> & interactions = model.getInteractions(ID);
58  const std::vector<Model::Regulation> & regulations = model.getRegulations(ID);
59 
60  // Copy mask of the regulation and turn of tested interaction
61  std::vector<bool> other(regulations[regul_num].first);
62  other[inter_num] = false;
63 
64  // Cycle through regulations again until you find context just without current interaction
65  std::size_t regul_comp = 0;
66  while (true) {
67  // If context is missing
68  if (regul_comp >= regulations.size())
69  throw std::runtime_error("Not fount other complementary regulation for some regulation.");
70  // If context is found
71  if (regulations[regul_comp].first == other)
72  break;
73  regul_comp++;
74  }
75 
76  // Test observabilit
77  is_observable = is_observable | (subcolor[regul_num] != subcolor[regul_comp]);
78 
79  // Test if the requirements are satisfied, if not, return false
80  if (((interactions[inter_num].constrain == pos_cons) && (subcolor[regul_num] < subcolor[regul_comp]))
81  ||
82  ((interactions[inter_num].constrain == neg_cons) && (subcolor[regul_num] > subcolor[regul_comp])))
83  return false;
84 
85  return true;
86  }

96  bool testSubcolor (const SpecieID ID, const std::vector<std::size_t> & subcolor) const {
97  // get referecnces to Specie data
98  const std::vector<Model::Interaction> & interactions = model.getInteractions(ID);
99  const std::vector<Model::Regulation> & regulations = model.getRegulations(ID);
100 
101  // Cycle through interactions
102  for (std::size_t inter_num = 0; inter_num < interactions.size(); inter_num++) {
103  // Skip if there are no requirements
104  if (interactions[inter_num].constrain == none_cons)
105  continue;
106  bool is_observable = false;
107 
108  // Cycle through regulations and test constrains
109  for (std::size_t regul_num = 0; regul_num < regulations.size(); regul_num++) {
110  // Skip if the regulation does not contain requested interaction
111  if (!regulations[regul_num].first[inter_num])
112  continue;
113 
114  if(!testConstrains(is_observable, ID, regul_num, inter_num, subcolor))
115  return false;
116  }
117 
118  // Check observability
119  if (!is_observable && interactions[inter_num].observable)
120  return false;
121  }
122  return true;
123  }

134  void testColors(SpecieColors && valid, const SpecieID ID, const std::size_t colors_num, const std::vector<std::size_t> & bottom_color, const std::vector<std::size_t> & top_color) {
135  // Cycle through all possible subcolors for this specie
136  std::vector<std::size_t> subcolor(bottom_color);
137 
138  // Cycle through all colors
139  for (std::size_t subcolor_num = 0; subcolor_num < colors_num; subcolor_num++) {
140  // Test is it is feasieble, if so, save it
141  if (testSubcolor(ID, subcolor))
142  valid.push_back(subcolor);
143 
144  // Iterate subcolor
145  for (std::size_t regul_num = 0; regul_num < subcolor.size(); regul_num++) {
146  // Increase and end
147  if (subcolor[regul_num] < top_color[regul_num]) {
148  subcolor[regul_num]++;
149  break;
150  }
151  // Null and continue
152  else {
153  subcolor[regul_num] = bottom_color[regul_num];
154  }
155  }
156  }
157 
158  // Add computed subcolors
159  colors.push_back(std::move(valid));
160  }

174  const std::size_t getBoundaries(const SpecieID ID, std::vector<std::size_t> & bottom_color, std::vector<std::size_t> & top_color) {
175  // Obtain all regulations
176  auto regulations = model.getRegulations(ID);
177  std::size_t colors_num = 1;
178 
179  // Cycle through regulations
180  for (std::size_t regul_num = 0; regul_num < regulations.size(); regul_num++) {
181  // If the target value is parametrized, add all the values
182  if (regulations[regul_num].second < 0) {
183  bottom_color[regul_num] = model.getMin(ID);
184  top_color[regul_num] = model.getMax(ID);
185  colors_num *= (model.getMax(ID) + 1);
186  }
187  // Otherwise add just given value
188  else {
189  bottom_color[regul_num] = top_color[regul_num] = regulations[regul_num].second;
190  colors_num *= 1;
191  }
192  }
193 
194  return colors_num;
195  }

202  void createContexts(const SpecieID ID) {
203  // Data to fill
204  SpecieColors valid;
205  valid.ID = ID;
206 
207  // Reference data
208  auto regulations = model.getRegulations(ID);
209  std::size_t colors_num; // How many colors will be tested (number of all combinations)
210 
211  // Create boundaries for iteration
212  std::vector<std::size_t> bottom_color(regulations.size());
213  std::vector<std::size_t> top_color(regulations.size());
214  valid.colors_num = colors_num = getBoundaries(ID, bottom_color, top_color);
215 
216  // Test all the subcolors and save feasible
217  testColors(std::move(valid), ID, colors_num, bottom_color, top_color);
218  }
