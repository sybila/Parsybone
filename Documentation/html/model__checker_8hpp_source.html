00055         void passParameters(Paramset & passed, const std::size_t step_size, const std::vector<bool> & transitive_values) const {
00056                 // INITIALIZATION OF VALUES FOR POSITIONING
00057                 // Number of the first parameter
00058                 ColorNum param_num = synthesis_range.first;
00059                 // First value might not bet 0 - get it from current parameter position
00060                 std::size_t value_num = (param_num / step_size) % transitive_values.size();
00061                 // Mask that will be created
00062                 register Paramset temporary = 0;
00063 
00064                 // COMPUTATION OF MASK
00065                 // List through all the paramters
00066                 while (true) {
00067                         // List through ALL the target values
00068                         for (; value_num < transitive_values.size(); value_num++) {
00069                                 // Get size of the step for current value 
00070                                 std::size_t bits_in_step = std::min<std::size_t>(step_size, static_cast<std::size_t>(synthesis_range.second - param_num));
00071                                 // Move the mask so new value data can be add
00072                                 temporary <<= bits_in_step;
00073                                 // If transitive, add ones for the width of the step
00074                                 if (transitive_values[value_num]) {
00075                                         Paramset add = ~0;
00076                                         add >>= (paramset_helper.getParamsetSize() - bits_in_step);
00077                                         temporary |= add;
00078                                 }
00079                                 // If we went throught the whole size, end
00080                                 if ((param_num += bits_in_step) == synthesis_range.second) {
00081                                         // Create interection of source parameters and transition parameters
00082                                         passed &= temporary;
00083                                         return;
00084                                 }
00085                         }
00086                         // Reset the value
00087                         value_num = 0;
00088                 }
00089         }

00147         std::vector<Coloring> broadcastParameters(const StateID ID, const Paramset parameters) {
00148                 // To store parameters that passed the transition but were not yet added to the target
00149                 std::vector<Coloring> param_updates;
00150                 param_updates.reserve(product.getTransitionCount(ID));

00154 
00155                 // Cycle through all the transition
00156                 for (std::size_t trans_num = 0; trans_num < product.getTransitionCount(ID); trans_num++) {
00157                         // Parameters to pass through the transition
00158                         Paramset passed = parameters;
00159                         // From an update strip all the parameters that can not pass through the transition - color intersection on the transition
00160                         passParameters(passed, product.getStepSize(ID, trans_num), product.getTransitive(ID, trans_num));
00161 
00162                         StateID target_ID = product.getTargetID(ID, trans_num);
00163 
00164                         // Test if it is a possibility for a loop, if there is nothing outcoming, add to self-loop (if it is still possible)
00165          if (KS_state == product.getKSID(target_ID) ) {
00166             loop_params &= passed;
00167             if(loop_params) {
00168                StateID BA_ID = product.getBAID(target_ID);
00169                BA_presence[BA_ID] = true;
00170             }
00171                         }
00172                         // Else add normally and remove from the loop
00173                         else if (passed) {
00174                                 loop_params &= ~passed; // Retain only others within a loop
00175                                 param_updates.push_back(std::make_pair(target_ID, passed));
00176                         }
00177                 }       
00178 
00179       // If there is a self-loop, add it for all the BA states (its an intersection of transitional parameters for independent loops)
00180                 for(StateID BA_state = 0; BA_state < BA_presence.size(); BA_state++) {

00227         void doColoring() {
00228                 // While there are updates, pass them to succesing vertices
00229                 do  {
00230                         // Within updates, find the one with most bits
00231          StateID ID = getStrongestUpdate();
00232                         // Check if this is not the last round
00233          if (user_options.timeSeries()) {
00234                                 if (product.isFinal(ID))
00235                markLevels(storage.getColor(ID));
00236                                 else
00237                                         transferUpdates(ID, storage.getColor(ID) & restrict_mask);
00238                         }
00239          else
00240             transferUpdates(ID, storage.getColor(ID));
00241 
00242                         // Erase completed update from the set
00243                         updates.erase(ID);
00244 
00245          // If there this round is finished, but there are still paths to find
00246          if (updates.empty() && to_find ) {
00247             updates = std::move(next_updates);
00248             storage.addFrom(next_round_storage);
00249             restrict_mask = to_find;
00250             BFS_level++; // Increase level
00251                         }
00252                 } while (!updates.empty());
00253 
00254                 // After the coloring, pass cost to the coloring (and computed colors = starting - not found)           
00255       if (user_options.timeSeries())
00256          storage.setResults(BFS_reach, ~to_find & starting);
00257       else if (starting_state != ~static_cast<std::size_t>(0))
00258          storage.setResults(storage.getColor(starting_state));
