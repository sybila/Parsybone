53  PerColorStorage(const ColoringAnalyzer & _analyzer, const ColorStorage & _storage, const ProductStructure & _product) : analyzer(_analyzer), storage(_storage), product(_product) {
54  auto colors = analyzer.getColors();
55 
56  states.resize(product.getStateCount());
57  for (StateID ID = 0; ID < product.getStateCount(); ID++) {
58  states[ID].ID = ID;
59  }
60 
61  // Go through colors
62  for (auto color_it = colors.begin(); color_it != colors.end(); color_it++) {
63 
64  // Get round values
65  std::size_t color_num = getBitNum(color_it->first);
66  Parameters color_bit = color_it->first;
67 
68  for (auto state_it = states.begin(); state_it != states.end(); state_it++) {
69  ColorData color_data;
70  color_data.color_bit = color_bit;
71  color_data.color_num = color_num;
72  color_data.successors = std::move(storage.getNeighbours(state_it->ID, true, color_bit));
73  color_data.predecessors = std::move(storage.getNeighbours(state_it->ID, false, color_bit));
74  state_it->color_data.push_back(std::move(color_data));
75  }
76  }
77  }
