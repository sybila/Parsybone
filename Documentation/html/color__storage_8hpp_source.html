83  void addFrom(const ColorStorage & other) {
84  // For all states
85  auto this_state_it = states.begin();
86  for (StateID ID = 0; ID < states.size(); ID++, this_state_it++) {
87  // Copy params
88  this_state_it->parameters |= other.getColor(ID);
89 
90  // Copy succesors parameters
91  auto this_succ_it = this_state_it->successors.begin();
92  std::vector<Parameters> other_succs = std::move(other.getMarking(ID, true));
93  for (auto other_succ_it = other_succs.begin(); other_succ_it != other_succs.end(); this_succ_it++, other_succ_it++) {
94  *this_succ_it |= *other_succ_it;
95  }
96 
97  // Copy predecessors parameters
98  auto this_pred_it = this_state_it->predecessors.begin();
99  std::vector<Parameters> other_preds = std::move(other.getMarking(ID, false));
100  for (auto other_pred_it = other_preds.begin(); other_pred_it != other_preds.end(); this_pred_it++, other_pred_it++) {
101  *this_pred_it |= *other_pred_it;
102  }
103  }

117  inline bool update(const Parameters parameters, const StateID ID) {

200  inline const Neighbours getNeighbours(const StateID ID, const bool successors, const Parameters color_mask = ~0) const {
201  // reference
202  auto neigbours = successors ? states[ID].successors : states[ID].predecessors;
203  // Data to fill
204  Neighbours color_neigh;

206  // Add these from the color

208  forEach(neigbours, [&neigh_num, &color_neigh, color_mask](const Parameters neighbour) {
209  // Test if the color is present
