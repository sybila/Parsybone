44  std::vector<std::vector<std::size_t> > getSourceValues(const std::vector<Model::Interaction> & interactions, const std::vector<bool> & mask) const {
45  // Data to return
46  std::vector<std::vector<std::size_t> > source_values;
47 
48  // For each regulating interaction pass the information
49  for (std::size_t interaction_num = 0; interaction_num < interactions.size(); interaction_num++) {
50  std::vector<std::size_t> possible_levels; // In wich levels the regulating specie has to be for regulation to be active?
51 
52  // If regulation has to be active, store values from treshold above
53  if (mask[interaction_num] == true)
54  for (std::size_t possible_level = interactions[interaction_num].threshold; possible_level <= model.getMax(interactions[interaction_num].source); possible_level++)
55  possible_levels.push_back(possible_level);
56  // Otherwise store levels below the treshold
57  else
58  for (std::size_t possible_level = 0; possible_level < interactions[interaction_num].threshold; possible_level++)
59  possible_levels.push_back(possible_level);
60 
61  // Store computed values
62  source_values.push_back(std::move(possible_levels));
63  }
64 
65  return source_values;
66  }

77  // Data to return

93  void addRegulations(const SpecieID ID, std::size_t & step_size) const {
94  // get referecnces to Specie data
95  const std::vector<Model::Interaction> & interactions = model.getInteractions(ID);
96  const std::vector<Model::Regulation> & regulations = model.getRegulations(ID);
97 
98  // Go through regulations of a specie - each represents a single function
99  std::size_t regul_num = 0;
100  for (auto regul_it = regulations.begin(); regul_it != regulations.end(); regul_it++, regul_num++) {
101  // Compute allowed values for each regulating specie for this function to be active
102  std::vector<std::vector<std::size_t> > source_values = std::move(getSourceValues(interactions, regul_it->first));
103 
104  // Add target values (if input negative, add all possibilities), if positive, add current requested value
105  std::vector<std::size_t> possible_values = std::move(constrains.getTargetVals(ID, regul_num));
106  // std::vector<std::size_t> possible_values = std::move(computePossibleValues(regul_it->second, ID));
107 
108  // pass the function to the holder.
109  functions_structure.addRegulatoryFunction(ID, step_size, std::move(possible_values), std::move(source_values));
110  }
111  // Increase step size for the next function
112  step_size *= constrains.getTargetVals(ID, 0).size();
113  }

139  std::vector<std::size_t> getSourceSpecies(const std::size_t specie_ID) const {
140  // Storage
141  std::vector<std::size_t> source_species;
142  // Get reference
143  const std::vector<Model::Interaction> & interactions = model.getInteractions(specie_ID);
144  // Add all the values between 0 and max
145  for (auto inter_it = interactions.begin(); inter_it != interactions.end(); inter_it++)

167  void buildFunctions() {
168  output_streamer.output(stats_str, "Costructing Regulatory functions for: ", OutputStreamer::no_newl)
169  .output(model.getSpeciesCount(), OutputStreamer::no_newl).output(" species.");
170 
171  std::size_t step_size = 1; // Variable necessary for encoding of colors
172 
173  // Cycle through all the species
174  for (std::size_t specie_num = 0; specie_num < model.getSpeciesCount(); specie_num++) {
175 
176  // Compute data
177  std::vector<std::size_t> possible_values = std::move(getPossibleValues(specie_num));
178  std::vector<std::size_t> source_species = std::move(getSourceSpecies(specie_num));
179 
180  // Add specie
181  functions_structure.addSpecie(model.getName(specie_num), specie_num, std::move(possible_values), std::move(source_species));
182 
183  // Add regulations for this specie
184  addRegulations(specie_num, step_size);
185  }
186 
187  // Set the number by what would be step size for next function
188  functions_structure.parameter_count = step_size;
189  }
