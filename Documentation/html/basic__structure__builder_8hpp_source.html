13 // BasicStructureBuilder creates the BasicStructure (Simple Kripke Structure) from the model data.
14 // In each iteration of the creation, a new state is generated as a cartesian product of values of the species. All the combinations are used.
15 // Each state is provided with indexes of their neighbours. For each dimension (specie) there are three neighbours base on the change of the specie's value - up, stay or donw.

46  void storeNeigbours(const StateID ID, const Levels & state_levels, const Levels & maxes) {
47  for (std::size_t specie = 0; specie < species_count; specie++) {
48  // If this value is not the lowest one, add neighbour with lower
49  if (state_levels[specie] > 0)
50  structure.addNeighbour(ID, ID - index_jumps[specie], specie, down_dir);
51 
52  // Add yourself
53  structure.addNeighbour(ID, ID, specie, stay_dir);
54 
55  // If this value is not the highest one, add neighbour with higher
56  if (state_levels[specie] < maxes[specie])
57  structure.addNeighbour(ID, ID + index_jumps[specie], specie, up_dir);
58  }
59  }

67  Levels getNextState(Levels state_levels) const {
68  // ID of specie to iterate
69  int position = 0;
70  // Iterate through species
71  for (auto specie_level = state_levels.begin(); specie_level != state_levels.end(); specie_level++) {
72  if (*specie_level == maxes[position]) { // Is max? Set to 0 and continue rights.
73  (*specie_level) = 0;
74  position++;
75  }
76  else { // Else increase and return
77  (*specie_level)++;
78  return state_levels;
79  }
80  }
81  // All-zero is returned after last round
82  return state_levels;
83  }

91  void computeBoundaries() {
92  states_count = 1;
93  for(std::size_t specie_num = 0; specie_num < species_count; specie_num++) {
94  // Maximal values of species
95  maxes.push_back(model.getMax(specie_num));
96  // How many states
97  states_count *= (model.getMax(specie_num) + 1);
98  }
99  }
100 
106  void computeJumps() {
107  index_jumps.resize(species_count);
108  // How many far away are two neighbour in the vector
109  std::size_t jump_lenght = 1;
110  // Species with higher index cause bigger differences
111  for (std::size_t specie_num = 0; specie_num < species_count; specie_num++) {
112  index_jumps[specie_num] = jump_lenght;
113  jump_lenght *= (model.getMax(specie_num) + 1);
114  }
115  }

135  void buildStructure() {
136  output_streamer.output(stats_str, "Computing Kripke structure states, total number of states: ", OutputStreamer::no_newl)
137  .output(states_count, OutputStreamer::no_newl).output(".");
138 
139  // Create initial state (by its values)
140  Levels state_levels(species_count, 0);
141 
142  // Create states
143  for(std::size_t state_num = 0; state_num < states_count; state_num++) {
144  // Fill the structure with the state
145  structure.addState(state_num, state_levels);
146  storeNeigbours(state_num, state_levels, maxes);
147  // Generate new state for the next round
148  state_levels = std::move(getNextState(state_levels));
149  }
150  }
