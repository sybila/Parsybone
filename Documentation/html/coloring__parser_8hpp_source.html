52  void openFile(const std::string filename) {
53  input_file.open(filename, std::ios::in | std::ios::binary | std::ios::ate);
54  if (input_file.fail())
55  throw std::runtime_error(std::string("Failed to open input coloring mask file: ").append(filename).c_str());
56  file_size = input_file.tellg();
57  if (file_size % sizeof(Parameters) != 0)
58  throw std::runtime_error("Bitmask file has incorrect number of bits - it must be dividable by the size of Paramset.");
59  if (file_size / sizeof(Parameters) > colors_sets.max_size())
60  throw std::runtime_error("Bitmask is bigger than a possible size of the vector, can not be used due to memory boundaries.");
61  input_file.seekg(0, std::ios::beg);
62  input_mask = true;
63  }

80  void parseMask() {
81  // Read bytemaks
82  char * byteblock = new char [static_cast<std::size_t>(file_size)];
83  input_file.read(byteblock, file_size);
84 
85  // Cycle through bytemasks and reforge them into Parameters
86  for (std::size_t shade_num = 0; shade_num < file_size / sizeof(Parameters); shade_num++) {
87  Parameters temp = 0;
88  // Read bytes and move them to the front of the Parameters, when possible
89  for (std::size_t byte_num = 0; byte_num < sizeof(Parameters); byte_num++) {
90  temp <<= 8;
91  char val = byteblock[shade_num*sizeof(Parameters) + byte_num];
92  temp += static_cast<unsigned char>(val);
93  }
94  colors_sets.push_back(temp);
95  }
96 
97  input_file.close();
98  }

105  void outputComputed(const Parameters parameters) {
106  // Cycle through bytes
107  for (std::size_t byte_num = sizeof(parameters); byte_num > 0; byte_num--) {
108  // Take last 8 bits - to save the form, it is needed to shift up till last round
109  output_file << static_cast<unsigned char>(parameters >> (byte_num-1)*8);
110  }
111  }
