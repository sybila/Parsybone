61  void eraseForbiddenValues(std::set<std::size_t> & current_values, const char op, const bool negate, const std::size_t compare_value, const std::size_t specie_ID) const {
62  // List through the original values and test if they satisfy the property, if not, remove them
63  std::for_each(all_values[specie_ID].begin(), all_values[specie_ID].end(), [op, negate, compare_value, &current_values](const std::size_t val) {
64  if (negate) { // Erase values satisfying the condition (not satisfying its negation)
65  if (op == '<' && val < compare_value )
66  current_values.erase(val);
67  else if ( op == '>' && val > compare_value )
68  current_values.erase(val);
69  else if ( op == '=' && val == compare_value )
70  current_values.erase(val);
71  }
72  else { // Erase values not satisfying the condition (operators are complementary)
73  if (op == '<' && val >= compare_value )
74  current_values.erase(val);
75  else if ( op == '>' && val <= compare_value )
76  current_values.erase(val);
77  else if ( op == '=' && val != compare_value )
78  current_values.erase(val);
79  }
80  });

90  AllowedValues applyConstrains(std::vector<std::string> & subformulas) const {
91  // Data to return
92  AllowedValues allowed_values(all_values);
93 
94  // Cycle through atomic formulas and get ALL the data
95  for (auto formula = subformulas.begin(); formula != subformulas.end(); formula++) {
96 
97  // Check negativity and delete the ! if needed
98  bool negate = ((*formula)[0] == '!');
99  if (negate) *formula = formula->substr( 1, formula->size());
100 
101  // Find position of the comparation operator
102  const std::size_t op_position = formula->find_first_of("<>=");
103 
104  // Specie ID
105  std::string specie_name = formula->substr(0, op_position);
106  const std::size_t ID = model.findID(specie_name);
107 
108  // Operator
109  char op = (*formula)[op_position]; // comparation operator
110 
111  // Value that is compared
112  std::size_t compare_value;
113  try {
114  compare_value = boost::lexical_cast<std::size_t,std::string>(formula->substr(op_position+1, formula->size()));
115  } catch (boost::bad_lexical_cast e) {
116  output_streamer.output(error_str, std::string("Error occured while parsing a label. It seems that you have entered non-numerical value as a value. ").append(e.what()));
117  throw std::runtime_error("boost::lexical_cast<size_t, std::string>(formula.substr(op_position+1, formula.size() - 1)) failed");
118  }
119 
120  // Erase the forbidden values for this atomic formula
121  eraseForbiddenValues(allowed_values[ID], op, negate, compare_value, ID);
122  }
123  return allowed_values;
124  }

133  AllowedValues parseConstrains(std::string constrains) const {
134  // If the label is always true or alway false, do not even bother with constrains and return all / none
135  if (constrains.compare("tt") == 0) {
136  AllowedValues allowed_values(all_values);
137  return std::move(allowed_values);
138  }
139  else if (constrains.compare("ff") == 0) {
140  AllowedValues none;
141  return std::move(none);
142  }
143 
144  // Check if all the characters are within allowed (nums, alpha, <, >, =, !, &)
145  std::for_each(constrains.begin(), constrains.end(), [&constrains](char ch) {
146  if (!(isalpha(ch) || isdigit(ch) || ch == '<' || ch == '>' || ch == '=' || ch == '!' || ch == '&'))
147  throw std::runtime_error(std::string("String: ").append(constrains).append(" contains invalid character: ").append(&ch).c_str());
148  });
149 
150  // Get all subparts of dual clause (splitting)
151  std::vector<std::string> subformulas;
152  try {
153  boost::split(subformulas, constrains, boost::is_any_of("&"));
154  } catch (std::exception & e) {
155  output_streamer.output(error_str, std::string("Error occured while parsing a label. ").append(e.what()));
156  throw std::runtime_error("boost::split(subformulas, constrains, boost::is_any_of(\"&\")) failed");
157  }
158 
159  // Return values with
160  return std::move(applyConstrains(subformulas));
161  }

176  void addTransitions(const StateID ID) const {
177  const std::vector<Model::Egde> & edges = model.getEdges(ID);
178 
179  // Transform each edge into transition and pass it to the automaton
180  for (std::size_t edge_num = 0; edge_num < model.getEdges(ID).size(); edge_num++) {
181  // Compute allowed values from string of constrains
182  AllowedValues constrained_values = std::move(parseConstrains(edges[edge_num].second));
183  // If the transition is possible for at least some values, add it
184  if (!constrained_values.empty())
185  automaton.addTransition(ID, edges[edge_num].first, std::move(constrained_values));
186  }
187  }
